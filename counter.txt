    <h1>Counter App</h1>
    <div id="count">0</div>

    <button onclick="decrement()">- Decrease</button>
    <button onclick="increment()">+ Increase</button>
    <button onclick="reset()">Reset</button>

    <script>
        let count = 0;

        function increment() {
            count++;
            document.getElementById("count").textContent = count;
        }

        function decrement() {
            count--;
            document.getElementById("count").textContent = count;
        }

        function reset() {
            count = 0;
            document.getElementById("count").textContent = count;
        }
    </script>
